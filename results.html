Introduction

This project replicates and extends Rees and Schnepel’s study on whether college football games are associated with increases in crime in host communities. We focus on daily assault and vandalism counts from NIBRS and match them to college football schedules from 2000 to 2005.

appendix_long <- analysis_data_clean %>%
  mutate(

      home_game == 1 ~ "Home Game",
      away_game == 1 ~ "Away Game",
      TRUE ~ "No Game"
    )

  pivot_longer(
    cols = c(assault, vandalism),

    values_to = "count"
  )

appendix_table1 <- bind_rows(
  appendix_long %>% mutate(game_type = "All Days"),
  appendix_long
) %>%
  group_by(crime_type, game_type) %>%
  summarise(
    mean_sd = paste0(round(mean(count), 2), " (", round(sd(count), 2), ")"),
    q25 = as.character(quantile(count, 0.25)),
    q50 = as.character(quantile(count, 0.50)),
    q75 = as.character(quantile(count, 0.75)),
    q90 = as.character(quantile(count, 0.90)),
    .groups = "drop"
  ) %>%
  pivot_longer(
    cols = c(mean_sd, q25, q50, q75, q90),
    names_to = "stat",
    values_to = "value"
  ) %>%
  mutate(
    stat = recode(
      stat,
      mean_sd = "",
      q25 = "0.25 quantile",
      q50 = "0.50 quantile",
      q75 = "0.75 quantile",
      q90 = "0.90 quantile"
    ),
    game_type = factor(
      game_type,
      levels = c("All Days", "No Game", "Home Game", "Away Game")
    ),
    crime_type = recode(
      crime_type,
      assault = "Assaults",
      vandalism = "Vandalism"
    )
  ) %>%
  arrange(crime_type, stat, game_type) %>%
  pivot_wider(
    names_from = game_type,
    values_from = value
  ) %>%
  select(
    crime_type,
    stat,
    `All Days`,
    `No Game`,
    `Home Game`,
    `Away Game`
  )

crime_daily <- read_csv("data/processed/crime_daily.csv", show_col_types = FALSE)
football_games <- read_csv("data/processed/football_games.csv", show_col_types = FALSE)
team_agency <- read_csv("data/processed/team_agency_mapping.csv", show_col_types = FALSE)

team_dates <- team_agency %>%
  crossing(date = seq(as.Date("2000-01-01"), as.Date("2005-12-31"), by = "day"))

home_games <- football_games %>%
  inner_join(team_agency, by = c("home_team" = "team")) %>%
  transmute(
    team = home_team,
    ori,
    date = game_date,
    home_game = 1,
    away_game = 0,
    home_win = home_win
  )

away_games <- football_games %>%
  inner_join(team_agency, by = c("away_team" = "team")) %>%
  transmute(
    team = away_team,
    ori,
    date = game_date,
    home_game = 0,
    away_game = 1,
    home_win = NA
  )

games_team_day <- bind_rows(home_games, away_games) %>%
  group_by(team, ori, date) %>%
  summarise(
    home_game = max(home_game),
    away_game = max(away_game),
    home_win = ifelse(any(home_game == 1), home_win[home_game == 1][1], NA),
    .groups = "drop"
  )

analysis_data <- team_dates %>%
  left_join(crime_daily, by = c("ori", "date" = "incident_date")) %>%
  left_join(games_team_day, by = c("team", "ori", "date")) %>%
  mutate(
    assault = replace_na(assault, 0),
    vandalism = replace_na(vandalism, 0),
    home_game = replace_na(home_game, 0),
    away_game = replace_na(away_game, 0),
    year = year(date),
    month = month(date),
    day_of_week = wday(date, label = TRUE)
  )

write_csv(analysis_data, "data/processed/analysis_data.csv")

team_summary_home <- analysis_data %>%
  filter(day_of_week == "Sat") %>%
  mutate(
    game_type = case_when(
      home_game == 1 ~ "Home Game",
      home_game == 0 & away_game == 0 ~ "No Game",
      TRUE ~ NA_character_
    )
  ) %>%
  filter(!is.na(game_type)) %>%
  group_by(team, game_type) %>%
  summarise(
    mean_assault = mean(assault),
    mean_vandalism = mean(vandalism),
    .groups = "drop"
  ) %>%
  group_by(team) %>%
  filter(
    n_distinct(game_type) == 2,
    max(mean_assault, mean_vandalism) > 0
  ) %>%
  ungroup()
